 package com.redhat.catalog.domain_model.publisher;
 
 import com.redhat.catalog.domain_model.book.isbn13.Publication;
 
 import java.util.Objects;
 
 /**
  * 【集約】Publisher クラス.
  *
  * @author kkosugi
  */
 public final class Publisher {
 
     private final Publication publication;
 
     private final PublisherName publisherName;
 
     /**
      * コンストラクタ.
      *
      * @param publication   出版社の ID を表す Publication インスタンス
      * @param publisherName 出版社の名前
      * @see Publication
      */
     public Publisher(Publication publication, PublisherName publisherName) {
         // null チェック
         if(Objects.isNull(publication) || Objects.isNull(publisherName)) {
             throw new IllegalArgumentException("Publication, PublisherName には null を設定することはできません.");
         }
 
         this.publication = publication;
         this.publisherName = publisherName;
     }
 
     @Override
     public String toString() {
         return "Publisher[" +
                 "publication=" + publication +
                 ", publisherName=" + publisherName +
                 ']';
     }
 }
