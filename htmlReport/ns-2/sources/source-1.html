 /**
  * Converter クラス.
  *
  * @author kkosugi

 public final class Converter {
 
     private Converter() {}
 
     /**
      * BookDAO インスタンスを Book インスタンスに変換して返却
      *
      * @param bookDAO ORM によって生成された BookDAO インスタンス
      * @return 変換された Book インスタンス
      */
     public static Book convertFromBookDao(BookDAO bookDAO) {
         // null チェック
         if(Objects.isNull(bookDAO)) {
             throw new IllegalArgumentException("bookDAO には null を設定することはできません.");
         }
 
         // String 型から ISBN13 型への変換
         var isbn13 = ISBN13.convertFromString(bookDAO.getIsbn13());
 
         // book の生成
         return new Book(isbn13, new Title(bookDAO.getTitle()), convertFromAuthorDao(bookDAO.getAuthorList()),
                 new PublisherName(bookDAO.getPublisher().getPublisherName()), new Price(bookDAO.getPrice()));
     }

         return null;
     }
