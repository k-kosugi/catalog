 package com.redhat.catalog.domain_model.book;
 
 /**
  * Price レコード.
  *
  * @param value 書籍の価格
  */
 public record Price(int value) {
 
     public static int MIN_VALUE = 0;
 
     /**
      * コンストラクタ.
      *
      * @param value 書籍の価格
      */
     public Price {

         }
     }
 
 }
