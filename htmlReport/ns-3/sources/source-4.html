 package com.redhat.catalog.domain_model.author;
 
 import java.util.Objects;
 
 /**
  * Author レコード.
  *
  * @param firstName 著者の姓
  * @param lastName  著者の名
  */
 public record AuthorName(Name firstName, Name lastName) {
 
     public static final String SEPARATOR = " ";
 
     /**
      * コンストラクタ
      *
      * @param firstName 著者の姓
      * @param lastName  著者の名
      */
     public AuthorName {
         // null チェック
         if (Objects.isNull(firstName) || Objects.isNull(lastName)) {
             throw new IllegalArgumentException("firstName と lastName には null を設定することができません.");
         }
     }
 
     /**
      * 著者のフルネームを firstName, lastName から返却します.
      *
      * @return 著者のフルネーム
      */
     protected String fullName() {
         return this.lastName.value() + SEPARATOR + this.firstName.value();
     }
 
     @Override
     public String toString() {
         return "AuthorName[" +
                 "firstName=" + firstName +
                 ", lastName=" + lastName +
                 ']';
     }
 }
